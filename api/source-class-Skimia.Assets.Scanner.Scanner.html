 96     protected function getOrderedFileDefinitions()
 97     {
 98         $files_defs = [];
 99         foreach ($this->directories as $path) {
100             if (! \File::exists($path)) {
101                 continue;
102             }
103 
104             $finder = Finder::create()->files()->ignoreDotFiles(false)->in($path);
105 
106 
107             $max_depth = $this->app['config']->get('assets.max_depth',3);
108             if(isset($this->directories_options[$path]['max_depth']))
109                 $max_depth = $this->directories_options[$path]['max_depth'];
110 
111             $files = $finder->depth('<= ' . $max_depth)->name('.assets.json');
112 
113 
114             foreach ($files as $file) {
115                 $content = $this->filterFile(json_decode($file->getContents(), true));
116                 $content['__dir'] = dirname($file->getRealpath());
117                 $files_defs[isset($content['alias']) ? $content['alias']:$content['name']] = $content;
118             }

277     protected function dep_resolve($list, $node, &$resolved, &$unresolved)
278     {
279         //echo $node['name']."<br/>";
280         $unresolved[] = $node;
281         if(isset($node['require']))
282         foreach ($node['require'] as $required) {
283             if(!isset($list[$required]))
284                 throw new \Exception('Unknown or not accessible dep: '.$required.' for '.$node['name'].( isset($node['alias']) ?'('.$node['alias'].')':''));
285             if(!in_array($list[$required],$resolved)){
286 
287                 if(in_array($list[$required],$unresolved))
288                     throw new \Exception('Circular reference detected: '.$node['name'].' -> '.$list[$required]['name'].( isset($list[$required]['alias']) ?'('.$list[$required]['alias'].')':''));
289                 $this->dep_resolve($list,$list[$required],$resolved,$unresolved);
290             }
291         }
292         if (!isset($node['resolve_this']) || $node['resolve_this'] === true) {
293             $resolved[] = $node;
294         }
295         unset($unresolved[array_search($node, $unresolved)]);
296     }
